    WHERE ENAME LIKE 'M%';       -- % : 0글자 이상, _ : 한글자

-- EQUI JOIN (공통 필드의 DEPTNO값이 일치되는 조건만 JOIN)
SELECT *
    FROM EMP, DEPT
    WHERE ENAME = 'SCOTT' AND EMP.DEPTNO = DEPT.DEPTNO;


-- FROM 절에 별칭 사용 가능
-- 테이블에 별칭이 있으면 그 별칭으로만 액서스를 할수있다.

SELECT EMPNO NO, ENAME, E.DEPTNO, DNAME, LOC 
    FROM EMP E, DEPT D
    WHERE E.DEPTNO = D.DEPTNO
    ORDER BY NO;
